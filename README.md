build_target() {
    local os=$1
    local arch=$2
    local ext="" # –†–∞—Å—à–∏—Ä–µ–Ω–∏–µ —Ñ–∞–π–ª–∞ (–ø—É—Å—Ç–æ–µ –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é)

    # –ï—Å–ª–∏ —Å–æ–±–∏—Ä–∞–µ–º –ø–æ–¥ Windows, –¥–æ–±–∞–≤–ª—è–µ–º —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–µ .exe
    if [ "$os" = "windows" ]; then
        ext=".exe"
    fi

    echo "–°–æ–±–∏—Ä–∞–µ–º –¥–ª—è $os/$arch..."

    # –°–æ–∑–¥–∞–µ–º –ø–æ–¥–¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—é –¥–ª—è –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–π —Å–±–æ—Ä–∫–∏, –Ω–∞–ø—Ä–∏–º–µ—Ä, "build/forge-linux-amd64"
    local target_dir="$BUILD_DIR/forge-$os-$arch"
    mkdir -p $target_dir

    # –°–æ–±–∏—Ä–∞–µ–º –±–∏–Ω–∞—Ä–Ω—ã–π —Ñ–∞–π–ª –¥–µ–º–æ–Ω–∞ 'forged'.
    # -o: —Ñ–ª–∞–≥ –¥–ª—è —É–∫–∞–∑–∞–Ω–∏—è –ø—É—Ç–∏ –∏ –∏–º–µ–Ω–∏ –≤—ã—Ö–æ–¥–Ω–æ–≥–æ —Ñ–∞–π–ª–∞.
    GOOS=$os GOARCH=$arch go build -o "$target_dir/forged$ext" ./cmd/forged

    # –°–æ–±–∏—Ä–∞–µ–º –±–∏–Ω–∞—Ä–Ω—ã–π —Ñ–∞–π–ª –∫–ª–∏–µ–Ω—Ç–∞ 'forge'.
    GOOS=$os GOARCH=$arch go build -o "$target_dir/forge$ext" ./cmd/forge
}

# --- –¶–µ–ª–∏ –¥–ª—è —Å–±–æ—Ä–∫–∏ ---
# –ó–¥–µ—Å—å –º—ã –≤—ã–∑—ã–≤–∞–µ–º –Ω–∞—à—É —Ñ—É–Ω–∫—Ü–∏—é –¥–ª—è –∫–∞–∂–¥–æ–π –∫–æ–º–±–∏–Ω–∞—Ü–∏–∏ –û–°/–∞—Ä—Ö–∏—Ç–µ–∫—Ç—É—Ä—ã.

# –°–æ–±–∏—Ä–∞–µ–º –¥–ª—è Linux (64-bit)
build_target "linux" "amd64"

# –°–æ–±–∏—Ä–∞–µ–º –¥–ª—è Windows (64-bit)
build_target "windows" "amd64"

# –°–æ–±–∏—Ä–∞–µ–º –¥–ª—è macOS (Intel 64-bit)
build_target "darwin" "amd64"

# –°–æ–±–∏—Ä–∞–µ–º –¥–ª—è macOS (Apple Silicon ARM 64-bit)
build_target "darwin" "arm64"

echo ""
